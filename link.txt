#include <stdio.h>

int main() {
    int num1, num2;
    float num3, num4;

    // Read two integers from stdin
    scanf("%d %d", &num1, &num2);
    
    // Read two floating-point numbers from stdin
    scanf("%f %f", &num3, &num4);
    
    // Calculate and print the sum and difference of integers
    printf("%d %d\n", num1 + num2, num1 - num2);
    
    // Calculate and print the sum and difference of floating-point numbers rounded to one decimal place
    printf("%.1f %.1f\n", num3 + num4, num3 - num4);
    
    return 0;
}


#include <stdio.h>

int max_of_four(int a, int b, int c, int d) {
    int max = a; // Assume a is the maximum initially

    if (b > max) {
        max = b;
    }
    if (c > max) {
        max = c;
    }
    if (d > max) {
        max = d;
    }

    return max;
}

int main() {
    int a, b, c, d;
    // Read four integers from input
    scanf("%d %d %d %d", &a, &b, &c, &d);

    // Find and print the maximum of the four integers
    int result = max_of_four(a, b, c, d);
    printf("%d\n", result);

    return 0;
}



#include <stdio.h>

int main() {
    char ch;
    char s[100];
    char sentence[100];

    // Input for a character
    scanf("%c", &ch);
    // Input for a string (word)
    scanf("%s", s);
    // Input for a sentence (including spaces)
    scanf(" %[^\n]s", sentence);

    // Print the character
    printf("%c\n", ch);
    // Print the string
    printf("%s\n", s);
    // Print the sentence
    printf("%s\n", sentence);

    return 0;
}


#include <stdio.h>
int main() {
   int year;
   printf("Enter a year: ");
   scanf("%d", &year);

   // leap year if perfectly visible by 400
   if (year % 400 == 0) {
      printf("%d is a leap year.", year);
   }
   // not a leap year if visible by 100
   // but not divisible by 400
   else if (year % 100 == 0) {
      printf("%d is not a leap year.", year);
   }
   // leap year if not divisible by 100
   // but divisible by 4
   else if (year % 4 == 0) {
      printf("%d is a leap year.", year);
   }
   // all other years are not leap year
   else {
      printf("%d is not a leap year.", year);
   }

   return 0;
}
